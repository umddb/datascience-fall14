CREATE TABLE clicks (
    OwnerId INTEGER,
    AdId INTEGER,
    numClicks INTEGER,
    PRIMARY KEY (OwnerId, AdId)
);

CREATE TABLE impressions (
    OwnerId INTEGER REFERENCES clicks(OwnerId),
    AdId INTEGER,
    numImpressions INTEGER
